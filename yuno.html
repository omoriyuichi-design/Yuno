<h1>🚄 乗り物あてゲームDX（5問版） 🚁</h1>

<p>ヒントをもとに5問の乗り物を当てよう！</p>

<button onclick="startGame()">ゲーム開始</button>

 

<div id="game-area" style="display:none;">

  <div class="question-count" id="question-count"></div>

  <div id="hint-area" class="hint"></div>

 

  <input type="text" id="guess" placeholder="乗り物の名前を入力">

  <button onclick="checkAnswer()">答える</button>

 

  <div class="message" id="message"></div>

  <div class="score" id="score"></div>

<script>

// === 乗り物データ（ヒント＋画像URL） ===

const vehicles = {

  "車": { hints: ["タイヤが4つある", "道路を走る", "ハンドルで運転する"], img: "https://upload.wikimedia.org/wikipedia/commons/3/3e/Toyota_Aqua_20140405.jpg" },

  "自転車": { hints: ["ペダルをこぐ", "タイヤが2つ", "人の力で動く"], img: "https://upload.wikimedia.org/wikipedia/commons/7/7e/Bicycle_on_the_road.JPG" },

  "バス": { hints: ["たくさんの人を乗せる", "停留所に止まる", "大きい車"], img: "https://upload.wikimedia.org/wikipedia/commons/5/5f/City_bus_in_Tokyo.jpg" },

  "電車": { hints: ["線路の上を走る", "駅に止まる", "たくさんの人を運ぶ"], img: "https://upload.wikimedia.org/wikipedia/commons/9/9d/JR_East_E231_series_train.jpg" },

  "新幹線": { hints: ["日本でとても速い電車", "長距離を走る", "のぞみやはやぶさがある"], img: "https://upload.wikimedia.org/wikipedia/commons/8/8b/Shinkansen_N700_series.jpg" },

  "飛行機": { hints: ["空を飛ぶ", "翼がある", "滑走路が必要"], img: "https://upload.wikimedia.org/wikipedia/commons/3/3f/Boeing_747-400_JA8961.jpg" },

  "ヘリコプター": { hints: ["回転する羽で浮く", "救助や報道にも使われる", "空を飛ぶ"], img: "https://upload.wikimedia.org/wikipedia/commons/3/32/Aerospatiale_Super_Puma_Autriche.jpg" }

};

 

// === ゲーム変数 ===

let remaining = [];

let answer = "";

let hints = [];

let currentHint = 0;

let scoreCount = 0;

let questionNum = 0;

const totalQuestions = 5;

 

function startGame() {

  remaining = Object.keys(vehicles);

  scoreCount = 0;

  questionNum = 0;

  document.getElementById("game-area").style.display = "block";

  nextQuestion();

}

 

function nextQuestion() {

  questionNum++;

  if (questionNum > totalQuestions || remaining.length === 0) {

    endGame();

    return;

  }

 

  const randomIndex = Math.floor(Math.random() * remaining.length);

  answer = remaining.splice(randomIndex, 1)[0];

  hints = vehicles[answer].hints;

  currentHint = 0;

 

  document.getElementById("message").textContent = "";

  document.getElementById("vehicle-img").style.display = "none";

  document.getElementById("question-count").textContent = `第 ${questionNum} 問 / ${totalQuestions}`;

  document.getElementById("score").textContent = `スコア：${scoreCount}`;

  showHint();

}

 

function showHint() {

  const hintArea = document.getElementById("hint-area");

  if (currentHint < hints.length) {

    hintArea.textContent = `ヒント ${currentHint + 1}：${hints[currentHint]}`;

  } else {

    hintArea.textContent = "ヒントはここまで！😅";

  }

}

 

function checkAnswer() {

  const guess = document.getElementById("guess").value.trim();

  const message = document.getElementById("message");

 

  if (guess === "") return;

 

  if (guess === answer) {

    message.textContent = `🎉 正解！「${answer}」です！`;

    message.style.color = "green";

    scoreCount++;

    document.getElementById("score").textContent = `スコア：${scoreCount}`;

    

    // 正解時に画像を表示

    const img = document.getElementById("vehicle-img");

    img.src = vehicles[answer].img;

    img.style.display = "block";

 

    setTimeout(nextQuestion, 2500);

  } else {

    currentHint++;

    if (currentHint < hints.length) {

      message.textContent = "❌ 違うよ！もう1つヒント！";

      message.style.color = "red";

      showHint();

    } else {

      message.textContent = `😢 残念！正解は「${answer}」でした。`;

      message.style.color = "blue";

 

      // 正解時に画像を表示

      const img = document.getElementById("vehicle-img");

      img.src = vehicles[answer].img;

      img.style.display = "block";

 

      setTimeout(nextQuestion, 3000);

    }

  }

  document.getElementById("guess").value = "";

  document.getElementById("guess").focus();

}

 

function endGame() {

  const message = document.getElementById("message");

  const hintArea = document.getElementById("hint-area");

  hintArea.textContent = "";

 

  let rank = "";

  if (scoreCount === totalQuestions) rank = "🚀 完璧！乗り物博士！";

  else if (scoreCount >= 4) rank = "🏆 上級ドライバー";

  else if (scoreCount >= 2) rank = "🚗 中級チャレンジャー";

  else rank = "🛴 初心者ライダー";

 

  message.innerHTML = `🎯 全${totalQuestions}問終了！<br>あなたのスコアは <b>${scoreCount}</b> 点！<br>${rank}`;

  message.style.color = "#333";

}

</script>
